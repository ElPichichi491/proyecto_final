function move(n){
i+=n;
if(i>=4)i=0;
if(i<0)i=3;
document.getElementById('slides').style.transform=`translateX(-${i*100}%)`;
document.querySelectorAll('.dot').forEach((dot,idx)=>{
dot.className='dot'+(idx===i?' active':'');
});
}

function go(n){
i=n-1;
move(1);
}

function show(s){
document.getElementById('main').classList.add('hidden');
document.getElementById('fturno').classList.add('hidden');
document.getElementById('fcurso').classList.add('hidden');
document.getElementById('sturno').classList.add('hidden');
document.getElementById('scurso').classList.add('hidden');
document.getElementById('scontacto').classList.add('hidden');

if(s==='servicios'){
document.getElementById('main').classList.remove('hidden');
}else if(s==='contacto'){
document.getElementById('scontacto').classList.remove('hidden');
}else if(s==='turnos'){
document.getElementById('sturno').classList.remove('hidden');

}else if(s==='cursos'){
document.getElementById('scurso').classList.remove('hidden');

}
}

function toggle(t){
document.getElementById('bs').style.opacity=t==='s'?'1':'0.6';
document.getElementById('bc').style.opacity=t==='c'?'1':'0.6';
if(t==='s'){
document.getElementById('slist').classList.remove('hidden');
document.getElementById('clist').classList.add('hidden');
}else{
document.getElementById('slist').classList.add('hidden');

}
}

// Servicios
function pick(s){

}

}
}

}

}

function backC(){
document.getElementById('fcurso').classList.add('hidden');

}
